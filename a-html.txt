    viewMode: null,

    // Cookie helper functions
    getCookie(name) {
        const value = `; ${document.cookie}`;
        const parts = value.split(`; ${name}=`);
        if (parts.length === 2) return parts.pop().split(';').shift();
        return null;
    },

    setCookie(name, value, days = 365) {
        const expires = new Date(Date.now() + days * 864e5).toUTCString();
        document.cookie = `${name}=${value}; expires=${expires}; path=/; SameSite=Lax`;
    },

    getDefaultView() {
        return window.innerWidth &lt; 1024 ? 'list' : 'grid';
    },

    init() {
        // Priority: Cookie &gt; localStorage &gt; Responsive Default
        const cookieValue = this.getCookie('viewMode');
        const localValue = localStorage.getItem('viewMode');

        if (cookieValue) {
            this.viewMode = cookieValue;
            // Cookie varsa localStorage'ƒ± da g√ºncelle (senkronize)
            localStorage.setItem('viewMode', cookieValue);
        } else if (localValue) {
            this.viewMode = localValue;
            // localStorage varsa cookie'ye de yaz
            this.setCookie('viewMode', localValue);
        } else {
            // Hi√ßbiri yoksa responsive default
            this.viewMode = this.getDefaultView();
        }

        // viewMode deƒüi≈üikliklerini hem localStorage hem cookie'ye kaydet
        this.$watch('viewMode', value =&gt; {
            localStorage.setItem('viewMode', value);
            this.setCookie('viewMode', value);
        });
    }

document.addEventListener('alpine:init', () => {
    // Product card price component
    Alpine.data('productCard', (hasTryPrice = false) => ({
        priceHovered: false,
        showTryPrice: false,
        priceTimer: null,
        hasTryPrice: hasTryPrice,
        init() {
            if (this.hasTryPrice) {
                this.startPriceCycle();
            }
        },
        startPriceCycle() {
            this.priceTimer = setInterval(() => {
                if (!this.priceHovered) {
                    this.showTryPrice = true;
                    setTimeout(() => {
                        if (!this.priceHovered) {
                            this.showTryPrice = false;
                        }
                    }, 1500); // TL: 1.5 saniye
                }
            }, 4500); // D√∂ng√º: 4.5 saniye (USD 3s + TRY 1.5s)
        },
        destroy() {
            if (this.priceTimer) clearInterval(this.priceTimer);
        }
    }));

    // Add to cart button component
    Alpine.data('addToCartButton', (productId) => ({
        loading: false,
        success: false,
        async addToCart() {
            console.log('üõí Alpine: addToCart clicked', { productId });
            this.loading = true;

            try {
                // CSRF token
                const token = document.querySelector('meta[name="csrf-token"]')?.getAttribute('content');

                console.log('üõí Alpine: Sending request to /api/cart/add');

                // localStorage'dan cart_id al (session sorunu i√ßin)
                const storedCartId = localStorage.getItem('cart_id');

                const response = await fetch('/api/cart/add', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'X-CSRF-TOKEN': token,
                        'Accept': 'application/json',
                    },
                    body: JSON.stringify({
                        product_id: productId,
                        quantity: 1,
                        cart_id: storedCartId ? parseInt(storedCartId) : null  // cart_id g√∂nder
                    })
                });

                const data = await response.json();
                console.log('üõí Alpine: API Response', data);

                if (data.success) {
                    this.success = true;

                    // localStorage'a cart bilgilerini kaydet (session sorunu i√ßin)
                    localStorage.setItem('cart_item_count', data.data.item_count);
                    localStorage.setItem('cart_id', data.data.cart_id);
                    console.log('üíæ localStorage: cart_item_count =', data.data.item_count);

                    // CartWidget'ƒ± g√ºncelle - Livewire event dispatch
                    if (typeof Livewire !== 'undefined') {
                        Livewire.dispatch('cartUpdated');
                        console.log('üîî Alpine: Livewire.dispatch(cartUpdated) √ßaƒürƒ±ldƒ±');
                    }

                    // Browser event (Alpine.js i√ßin)
                    window.dispatchEvent(new CustomEvent('cart-updated', {
                        detail: {
                            itemCount: data.data.item_count,
                            cartId: data.data.cart_id
                        }
                    }));

                    console.log('‚úÖ Alpine: Success! Item count:', data.data.item_count);

                    setTimeout(() => { this.success = false; }, 2000);
                } else {
                    console.error('‚ùå Alpine: API returned error', data.message);
                    alert(data.message || '√úr√ºn sepete eklenirken hata olu≈ütu');
                }
            } catch (error) {
                console.error('‚ùå Alpine: Fetch error', error);
                alert('√úr√ºn sepete eklenirken hata olu≈ütu');
            } finally {
                this.loading = false;
            }
        }
    }));
});
