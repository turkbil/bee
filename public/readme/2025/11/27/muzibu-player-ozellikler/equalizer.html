class MuzibuEqualizer {
    constructor(audioContext) {
        this.audioContext = audioContext;
        this.bands = [];
        this.frequencies = [32, 64, 125, 250, 500, 1000, 2000, 4000, 8000, 16000];

        // 10 band filter olu≈ütur
        this.frequencies.forEach(freq => {
            const filter = this.audioContext.createBiquadFilter();
            filter.type = 'peaking'; // Parametric EQ
            filter.frequency.value = freq;
            filter.Q.value = 1.0; // Bandwidth (keskinlik)
            filter.gain.value = 0; // Ba≈ülangƒ±√ß: 0 dB (flat)

            this.bands.push({ frequency: freq, filter });
        });

        // Filter chain olu≈ütur (band'larƒ± sƒ±rayla baƒüla)
        for (let i = 0; i < this.bands.length - 1; i++) {
            this.bands[i].filter.connect(this.bands[i + 1].filter);
        }
    }

    // Input node (audio source baƒülanƒ±r)
    get input() {
        return this.bands[0].filter;
    }

    // Output node (destination'a baƒülanƒ±r)
    get output() {
        return this.bands[this.bands.length - 1].filter;
    }

    // Tek bir band'ƒ± ayarla
    setBand(index, gainDB) {
        if (index >= 0 && index < this.bands.length) {
            this.bands[index].filter.gain.value = gainDB;
        }
    }

    // T√ºm band'larƒ± ayarla (preset i√ßin)
    setAllBands(gainsArray) {
        gainsArray.forEach((gain, index) => {
            this.setBand(index, gain);
        });
    }

    // Reset (flat)
    reset() {
        this.bands.forEach(band => {
            band.filter.gain.value = 0;
        });
    }
}
