        function logoAnimation() {
            const logo = document.querySelector('.main-logo');
            const otherElements = document.querySelectorAll('.reveal:not(.main-logo)');

            // First show logo
            logo.classList.add('active');

            // After logo appears, slide it up and show others
            setTimeout(() => {
                logo.style.transform = 'translateY(-50px)';
                logo.style.transition = 'transform 0.8s ease-out';

                // Show other elements cascading down
                otherElements.forEach((element, index) => {
                    setTimeout(() => {
                        element.classList.add('active');
                    }, index * 150);
                });
            }, 800);
        }

            const reveals = document.querySelectorAll('.reveal');

                }, index * 200);
            });
        }

            const particle = document.createElement('div');

            particle.style.width = Math.random() * 6 + 4 + 'px';
            particle.style.height = particle.style.width;
            document.body.appendChild(particle);

            setTimeout(() => {
                particle.remove();
            }, 4000);
