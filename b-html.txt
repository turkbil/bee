// Form view tracking
gtag && gtag('event', 'page_view', {
    'page_title': 'Elektrikli Transpalet Landing',
    'page_location': window.location.href,
    'page_path': window.location.pathname

// Scroll tracking (50%, 75%, 100%)
let scrollTracked = {50: false, 75: false, 100: false};
window.addEventListener('scroll', () => {
    const scrollPercent = (window.scrollY / (document.documentElement.scrollHeight - window.innerHeight)) * 100;

    if (scrollPercent >= 50 && !scrollTracked[50]) {
        gtag && gtag('event', 'scroll', {'event_category': 'Engagement', 'event_label': '50%'});
        scrollTracked[50] = true;
    }
    if (scrollPercent >= 75 && !scrollTracked[75]) {
        gtag && gtag('event', 'scroll', {'event_category': 'Engagement', 'event_label': '75%'});
        scrollTracked[75] = true;
    }
    if (scrollPercent >= 99 && !scrollTracked[100]) {
        gtag && gtag('event', 'scroll', {'event_category': 'Engagement', 'event_label': '100%'});
        scrollTracked[100] = true;
    }
});
